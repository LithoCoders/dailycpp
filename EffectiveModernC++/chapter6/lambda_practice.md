#include <iostream>
#include <utility>
#include <map>
#include <string>
#include <algorithm>
#include <type_traits>

using namespace std;

using DBxItem = map<string, int>;


int main()
{
    DBxItem data;
    data.insert(make_pair("tuna", 14));
    data.insert(make_pair("salmon", 12));
    data.insert(make_pair("apple", 10));
    
    auto res = min_element(data.begin(), data.end(),
                            [](DBxItem::value_type & item1, DBxItem::value_type &item2) -> bool { return item1.second < item2.second; });
    cout << res->first << " " << res->second << endl;
    return 0;
}
